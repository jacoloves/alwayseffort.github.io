```rs
fn ddcc2020_qual_b() {
    input! {
        n: usize,
        a: [i64; n],
    }

    let mut s: i64 = 0;
    for i in 0..n {
        s += a[i];
    }

    let mut minx: i64 = s;
    let mut sum: i64 = 0;
    for i in 0..n {
        sum += a[i];
        minx = minx.min((sum - (s - sum)).abs());
    }

    println!("{}", minx);
}
```
